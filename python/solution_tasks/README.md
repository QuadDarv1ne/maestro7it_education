def calculate_average_grades(grades, students):
    # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –º–Ω–æ–∂–µ—Å—Ç–≤–æ –≤ —Å–ø–∏—Å–æ–∫ –¥–ª—è —É–¥–æ–±—Å—Ç–≤–∞ –¥–æ—Å—Ç—É–ø–∞
    students_list = list(students)

    # –°–æ–∑–¥–∞–µ–º –ø—É—Å—Ç–æ–π —Å–ª–æ–≤–∞—Ä—å –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è —Å—Ä–µ–¥–Ω–∏—Ö –±–∞–ª–ª–æ–≤
    student_grades = {}

    # –ü–µ—Ä–µ–±–∏—Ä–∞–µ–º –∫–∞–∂–¥–æ–≥–æ —É—á–µ–Ω–∏–∫–∞
    for student in students_list:
        # –ù–∞—Ö–æ–¥–∏–º –∏–Ω–¥–µ–∫—Å —É—á–µ–Ω–∏–∫–∞ –≤ —Å–ø–∏—Å–∫–µ students
        student_index = students_list.index(student)

        # –ü–æ–ª—É—á–∞–µ–º —Å–ø–∏—Å–æ–∫ –æ—Ü–µ–Ω–æ–∫ –¥–ª—è —Ç–µ–∫—É—â–µ–≥–æ —É—á–µ–Ω–∏–∫–∞
        student_scores = grades[student_index]

        # –í—ã—á–∏—Å–ª—è–µ–º —Å—Ä–µ–¥–Ω–∏–π –±–∞–ª–ª
        average_grade = sum(student_scores) / len(student_scores)

        # –î–æ–±–∞–≤–ª—è–µ–º –ø–∞—Ä—É "–∏–º—è-—Å—Ä–µ–¥–Ω–∏–π –±–∞–ª–ª" –≤ —Å–ª–æ–≤–∞—Ä—å
        student_grades[student] = average_grade

    # –í—ã–≤–æ–¥–∏–º —Å–ª–æ–≤–∞—Ä—å —Å—Ä–µ–¥–Ω–∏—Ö –±–∞–ª–ª–æ–≤
    return student_grades


# –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è —Ñ—É–Ω–∫—Ü–∏–∏
grades = [[5, 3, 3, 5, 4], [2, 2, 2, 3], [4, 5, 5, 2], [4, 4, 3], [5, 5, 5, 4, 5]]

print(calculate_average_grades(grades, students))
