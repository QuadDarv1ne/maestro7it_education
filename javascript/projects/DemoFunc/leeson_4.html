Примеры кода PHP

1. Работа с массивами

// Индексированный массив
$fruits = array("Яблоко", "Апельсин", "Банан");
// или
$fruits = ["Яблоко", "Апельсин", "Банан"];

// Ассоциативный массив
$age = array("Иван" => 25, "Мария" => 30, "Петр" => 35);

// Многомерный массив
$users = array(
    array("name" => "Иван", "age" => 25),
    array("name" => "Мария", "age" => 30),
    array("name" => "Петр", "age" => 35)
);

2. Циклы

// Цикл for
for ($i = 0; $i < 5; $i++) {
    echo "Номер: " . $i . "<br>";
}

// Цикл foreach для индексированного массива
foreach ($fruits as $fruit) {
    echo $fruit . "<br>";
}

// Цикл foreach для ассоциативного массива
foreach ($age as $name => $value) {
    echo $name . " - " . $value . " лет<br>";
}

// Цикл while
$i = 0;
while ($i < count($fruits)) {
    echo $fruits[$i] . "<br>";
    $i++;
}

3. Функции

// Объявление функции
function greet($name) {
    return "Привет, " . $name . "!";
}

// Вызов функции
echo greet("Иван"); // Выведет: Привет, Иван!

// Функция с параметром по умолчанию
function introduce($name, $city = "Москва") {
    return "Меня зовут " . $name . ", я из " . $city;
}

echo introduce("Анна"); // Выведет: Меня зовут Анна, я из Москва
echo introduce("Алексей", "Санкт-Петербург"); // Выведет: Меня зовут Алексей, я из Санкт-Петербург

4. Работа с формами

// HTML форма
<form method="POST" action="">
    Имя: <input type="text" name="name">
    Email: <input type="email" name="email">
    <button type="submit" name="submit">Отправить</button>
</form>

// Обработка формы в PHP
if (isset($_POST['submit'])) {
    $name = htmlspecialchars($_POST['name']);
    $email = htmlspecialchars($_POST['email']);
    
    // Валидация и обработка данных
    if (!empty($name) && !empty($email)) {
        echo "Привет, " . $name . "! Ваш email: " . $email;
    } else {
        echo "Пожалуйста, заполните все поля";
    }
}
